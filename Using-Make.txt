Unified Makefile quick reference

# List discovered examples
make list

# Build an example (default backend: PlatformIO)
make <Example>
make <Example> debug

# Upload (set your port)
make <Example>.upload PORT=/dev/ttyACM0
make <Example>.upload PORT=/dev/ttyACM0 PROTOCOL=wiring BAUD=115200

# Force backend
make -- --pio <Example>
make -- --arduino <Example>
make -- --avr <Example>

# Clean artifacts
make clean
make deepclean
